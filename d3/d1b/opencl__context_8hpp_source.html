101  opencl_context_base() {
102  try {
103  // platform
104  cl::Platform::get(&platforms_);
105  if (OPENCL_PLATFORM_ID >= platforms_.size()) {
106  system_error("OpenCL Initialization", "[Platform]", -1,
107  "CL_INVALID_PLATFORM");
108  }
109  platform_ = platforms_[OPENCL_PLATFORM_ID];
110  platform_name_ = platform_.getInfo<CL_PLATFORM_NAME>();
111  platform_.getDevices(DEVICE_FILTER, &devices_);
112  if (devices_.size() == 0) {
113  system_error("OpenCL Initialization", "[Device]", -1,
114  "CL_DEVICE_NOT_FOUND");
115  }
116  if (OPENCL_DEVICE_ID >= devices_.size()) {
117  system_error("OpenCL Initialization", "[Device]", -1,
118  "CL_INVALID_DEVICE");
119  }
120  device_ = devices_[OPENCL_DEVICE_ID];
121  // context and queue
122  context_ = cl::Context(device_);
123  command_queue_ = cl::CommandQueue(context_, device_,
124  CL_QUEUE_PROFILING_ENABLE, nullptr);
125  device_.getInfo<size_t>(CL_DEVICE_MAX_WORK_GROUP_SIZE,
126  &max_thread_block_size_);
127  int thread_block_size_sqrt
128  = static_cast<int>(sqrt(static_cast<double>(max_thread_block_size_)));
129  // Does a compile time check of the maximum allowed
130  // dimension of a square thread block size
131  // WG size of (32,32) works on all recent GPUs but would fail on some
132  // older integrated GPUs or CPUs
133  if (thread_block_size_sqrt < base_opts_["THREAD_BLOCK_SIZE"]) {
134  base_opts_["THREAD_BLOCK_SIZE"] = thread_block_size_sqrt;
135  base_opts_["WORK_PER_THREAD"] = 1;
136  }
137  if (max_thread_block_size_ < base_opts_["LOCAL_SIZE_"]) {
138  // must be a power of base_opts_["REDUCTION_STEP_SIZE"]
139  const int p = std::log(max_thread_block_size_)
140  / std::log(base_opts_["REDUCTION_STEP_SIZE"]);
141  base_opts_["LOCAL_SIZE_"]
142  = std::pow(base_opts_["REDUCTION_STEP_SIZE"], p);
143  }
144  // Thread block size for the Cholesky
145  // TODO(Steve): This should be tuned in a higher part of the stan language
146  if (max_thread_block_size_ >= 256) {
147  tuning_opts_.cholesky_min_L11_size = 256;
148  } else {
149  tuning_opts_.cholesky_min_L11_size = max_thread_block_size_;
150  }
151  } catch (const cl::Error& e) {
152  check_opencl_error("opencl_context", e);
153  }
154  }
